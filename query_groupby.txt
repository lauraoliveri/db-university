
1. Contare quanti iscritti ci sono stati ogni anno

SELECT year(enrolment_date), COUNT(*) AS 'newstudents'
FROM `students`
GROUP BY year(enrolment_date);

2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio

SELECT office_address, COUNT(*) AS 'teachers_with_same_office'
FROM `teachers`
GROUP BY office_address;


3. Calcolare la media dei voti di ogni appello d'esame

SELECT exam_id, ROUND(AVG(vote), 2) AS vote_avg
FROM `exam_student`
GROUP BY exam_id;

4. Contare quanti corsi di laurea ci sono per ogni dipartimento

SELECT department_id, COUNT(*) AS 'degrees_number'
FROM `degrees`
GROUP BY departments_id;
